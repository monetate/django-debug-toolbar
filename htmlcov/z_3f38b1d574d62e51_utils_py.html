13class ElideSelectListsFilter: 

14 """sqlparse filter to elide the select list from top-level SELECT ... FROM clauses, 

15 if present""" 

16

19 for token_type, value in stream: 

20 yield token_type, value 

21 if token_type in T.Keyword: 

22 keyword = value.upper() 

23 if allow_elision and keyword == "SELECT":

25 allow_elision = keyword in ["EXCEPT", "INTERSECT", "UNION"]
